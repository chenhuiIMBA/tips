    var wg sync.WaitGroup

        wg.Add(1)
        go func() {

            for job := range jobs {

        }()
    }

    close(results)
}

) {
  const [state, setState] = useState({
    data: null as T | null,
    loading: false,
    error: null as string | null
  })

    setState(prev => ({ ...prev, loading: true, error: null }))
